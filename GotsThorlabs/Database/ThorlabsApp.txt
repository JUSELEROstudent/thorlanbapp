CREATE TABLE user (
  idUser INTEGER PRIMARY KEY,
  name TEXT,
  lastname TEXT,
  telephon INTEGER,
  eMail TEXT,
  nickname TEXT,
  password TEXT
);

CREATE TABLE tour (
  idTour INTEGER PRIMARY KEY,
  date TEXT,
  nameFolder TEXT,
  NumberX INTEGER,
  NumberY INTEGER,
  NumberZ INTEGER,
  camera TEXT,
  endStatus TEXT,
  idStitching INTEGER,
  idUser INTEGER,
  FOREIGN KEY (idUser) REFERENCES user(idUser)
);

CREATE TABLE stitching (
  idStitching INTEGER PRIMARY KEY,
  path TEXT,
  idImgResult INTEGER,
  typestitching TEXT,
  status TEXT,
  idTour INTEGER,
  FOREIGN KEY (IdTour) REFERENCES tour(IdTour)
);

CREATE TABLE image (
  idImage INTEGER PRIMARY KEY,
  name TEXT,
  gausianVal INTEGER,
  path TEXT,
  X INTEGER,
  Y INTEGER,
  Z INTEGER,
  idTour INTEGER,
  FOREIGN KEY (idTour) REFERENCES tour(idTour)
);

CREATE TABLE logcalibrate (
  idLog INTEGER PRIMARY KEY,
  dateTimer TEXT,
  logCalibratedesc TEXT
);
